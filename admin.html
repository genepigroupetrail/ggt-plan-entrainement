🔐 Règles Firestore
À copier-coller dans Firebase → Firestore → Règles
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /plannings/{weekId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    match /vma_data/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    match /race_plans/{planId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    match /race_suivi/{docId} {
      allow read, write: if request.auth != null &&
        docId.matches(request.auth.uid + '_.*');
      allow read: if request.auth != null;
    }
  }
}
📅 Changement d'année — Navigation semaines
La navigation est configurée sur ±3120 semaines (60 ans) depuis aujourd'hui.
Pour étendre à une nouvelle année, modifier la valeur dans planning.html et admin.html : 
// Ligne à modifier dans loadWeek() / renderWeek()
document.getElementById('btn-prev').disabled = currentWeekOffset <= -3120;
document.getElementById('btn-next').disabled = currentWeekOffset >= 3120;
// Changer 3120 par 5200 pour 100 ans, etc.
